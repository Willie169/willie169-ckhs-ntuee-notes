\subsubsection{Euler's constant or EulerвЂ“Mascheroni constant}
Let
\[H_n=\sum_{n=1}^{\infty}\frac{1}{n}\]
\[t_n=H_n-\ln n\]
\[\lim_{n\to\infty}t_n=\int_1^{\infty}\frac{1}{\floor{x}}-\frac{1}{x}\dd{x}=\gamma\in\bbR,\]
which is called Euler's constant or EulerвЂ“Mascheroni constant and approximately 0.57721.
\bpr
\[t_n=\sum_{k=1}^n\frac{1}{k}-\sum_{k=1}^{n-1}\ln(k+1)-\ln k=\sum_{k=1}^{n-1}\int_k^{k+1}\frac{1}{k}-\frac{1}{x}\dd{x}+\frac{1}{n}=\int_1^n\frac{1}{\floor{x}}-\frac{1}{x}\dd{x}+\frac{1}{n}\]
\[\lim_{n\to\infty}\frac{1}{n}=0\]
\[\frac{1}{\floor{x}}-\frac{1}{x}\ge0\]
\[\frac{1}{n}>0\]
\[t_n>0\]
\[t_n-t_{n+1}=\ln(n+1)-\ln n-\frac{1}{n+1}=\int_n^{n+1}\frac{1}{x}-\frac{1}{n+1}\dd{x}\]
Since $\frac{1}{x}\ge\frac{1}{n+1}$ for $x\in[n,n+1]$,
\[\int_n^{n+1}\frac{1}{x}-\frac{1}{n+1}\dd{x}>0\]
$t_n$ is bounded below by 0, and $t_n$ is decreasing, hence $t_n$ is convergent.
\epr
It's unknown whether $\gamma$ is irrational.
\subsubsection{Alternating harmonic series convergence}
\[\sum_{n=1}^\infty\frac{(-1)^{n+1}}{n}=\ln 2.\]
\bpr
\[\ba
\ln 2&=\ln(1+1)\\
&=\int_0^1\frac{1}{1+x}\dd{x}\\
&=\int_0^1\sum_{n=0}^{\infty}(-x)^n\dd{x}\\
&=\sum_{n=0}^{\infty}\int_0^1(-x)^n\dd{x}\\
&=\sum_{n=0}^{\infty}(-1)^n\frac{1}{n+1}\\
&=\sum_{n=1}^\infty\frac{(-1)^{n+1}}{n}
\ea\]
\epr
\subsection{Riemann zeta function or EulerвЂ“Riemann zeta function}
\subsubsection{Definition}
The Riemann zeta function or EulerвЂ“Riemann zeta function $\zeta(s)$ is a complex function defined as the analytic extension of
\[\zeta(s)=\sum_{n=1}^\infty\frac{1}{n^s},\quad\forall\Re(s)>1\]
such that it is meromorphic and with an only pole at $s=1$.
