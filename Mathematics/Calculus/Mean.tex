\section{Mean}
\subsection{Abstract mean}
A map $M$ from a bounded function whose domain is a subset of $\bbR$ to a real number is an abstract mean iff for all $f,g$ that are bounded functions whose domains are subsets of $\bbR$ and $a,b\in\bbR$ such that both sides are defined,
\bit
\item Linearity:
\[M(af+bg)=aM(f)+bM(g).\]
\item Positivity:
\[f(x)\geq0\forall x\in D_f\implies M(f)\geq0.\]
\item Normalization:
\[f(x)=a\forall x\in D_f\implies M(f)=a.\]
\eit
\subsection{Arithmetic mean (AM)}
\subsubsection{Sequence}
For a finite sequence $\langle a_i\rangle_{i=l}^n$, the arithmetic mean $\bar{a}$ of it is defined as
\[\bar{a}=\frac{1}{n-l+1}\sum_{i=l}^na_i.\]
\subsubsection{Real-domain function}
For a function $f(x)$ defined on a nonempty interval $[a,b]\subseteq D_f$, the arithmetic mean $\bar{f}$ of it is defined as
\[\bar{f}=\frac{1}{b-a}\int_a^bf(x)\dd{x}.\]
\subsubsection{Measurable function}
For a measurable function $f$ defined on a measurable set $\Omega\subseteq D_f$ with $\mu(\Omega)\in\bbR$, the arithmetic mean $\bar{f}$ of it is defined as
\[\bar{f}=\frac{1}{\mu(\Omega)}\int_{\Omega}f\dd{\mu}.\]
\subsection{Geometric mean (GM)}
\subsubsection{Sequence}
For a finite sequence $\langle a_i\rangle_{i=l}^n$ of positive real number, the geometric mean of it is defined as
\[\exp\qty(\frac{1}{n-l+1}\sum_{i=l}^n\ln a_i).\]
\subsubsection{Real-domain function}
For a positive real-valued function $f(x)$ defined on a nonempty interval $[a,b]\subseteq D_f$, the geometric mean of it is defined as
\[\exp\qty(\frac{1}{b-a}\int_a^b\ln(f(x))\dd{x}).\]
\subsection{Harmonic mean (HM)}
\subsubsection{Sequence}
For a finite sequence $\langle a_i\rangle_{i=l}^n$ of nonzero number, the harmonic mean of it is defined as
\[\frac{n-l+1}{\sum_{i=l}^n\frac{1}{a_i}}.\]
\subsubsection{Real-domain function}
For a nonzero real-valued function $f(x)$ defined on a nonempty interval $[a,b]\subseteq D_f$, the harmonic mean of it is defined as
\[\frac{b-a}{\int_a^b\frac{1}{f(x)}\dd{x}}.\]
\subsection{Cesàro sum or Cesàro mean}
\subsubsection{Sequence}
For an infinite sequence $\langle a_i\rangle_{i=l}^{\infty}$, the Cesàro sum or Cesàro mean of it is defined as
\[\lim_{n\to\infty}\frac{1}{n}\sum_{i=l}^na_i.\]
If the Cesàro sum is finite, the sequence is called Cesàro-summable.
\subsubsection{Real-domain function}
For a function $f$ defined on $[a,\infty)$ for $a\in\bbR$ with $\int_a^bf(x)\dd{x}$ being a proper integral for all $b\in(a,\infty)$, the Cesàro sum or Cesàro mean of it is defined as
\[\lim_{n\to\infty}\frac{1}{n}\int_a^nf(x)\dd{x}.\]
If the Cesàro sum is finite, the function is called Cesàro-summable over $[a,\infty)$.

For a function $f$ defined on $(-\infty,b]$ for $b\in\bbR$ with $\int_a^bf(x)\dd{x}$ being a proper integral for all $a\in(-\infty,b)$, the Cesàro sum or Cesàro mean of it is defined as
\[\lim_{n\to\infty}\frac{1}{n}\int_{-n}^bf(x)\dd{x}.\]
If the Cesàro sum is finite, the function is called Cesàro-summable over $(-\infty,b]$.

For a function $f$ defined on $(-\infty,\infty)$ with $\int_a^bf(x)\dd{x}$ being a proper integral for all $a<b\land a,b\in\bbR$, the Cesàro sum or Cesàro mean of it is defined as
\[\lim_{n\to\infty}\frac{1}{2n}\int_{-n}^nf(x)\dd{x}.\]
If the Cesàro sum is finite, the function is called Cesàro-summable over $(-\infty,\infty)$.
\subsection{Root mean square (RMS or rms) or quadratic mean}
\subsubsection{Sequence}
For a finite sequence $\langle a_i\rangle_{i=l}^n$, the RMS $a_{rms}$ of it is defined as
\[a_{rms}=\sqrt{\frac{1}{n-l+1}\sum_{i=l}^na_i^{\pht{i}2}}.\]
For an infinite sequence $\langle a_i\rangle_{i=l}^{\infty}$, the RMS $a_{rms}$ of it is defined as
\[a_{rms}=\sqrt{\lim_{n\to\infty}\frac{1}{n-l+1}\sum_{i=l}^na_i^{\pht{i}2}}.\]
\subsubsection{Real-domain function}
For a function $f(x)$ defined on a nonempty interval $[a,b]$, the RMS $f_{rms}$ of it is defined as
\[f_{rms}=\sqrt{\frac{1}{b-a}\int_a^b(f(x))^2\dd{x}}.\]
For a function $f(x)$ defined on $[a,\infty)$ with $\int_a^bf(x)\dd{x}$ being a proper integral for all $b\in(a,\infty)$, the RMS $f_{rms}$ of it is defined as
\[f_{rms}=\sqrt{\lim_{b\to\infty}\frac{1}{b-a}\int_a^b(f(x))^2\dd{x}}.\]
For a function $f(x)$ defined on $(-\infty,b]$ with $\int_a^bf(x)\dd{x}$ being a proper integral for all $a\in(-\infty,b)$, the RMS $f_{rms}$ of it is defined as
\[f_{rms}=\sqrt{\lim_{a\to-\infty}\frac{1}{b-a}\int_a^b(f(x))^2\dd{x}}.\]
For a function $f(x)$ defined on $(-\infty,\infty)$ with $\int_a^bf(x)\dd{x}$ being a proper integral for all $a<b\land a,b\in\bbR$, the RMS $f_{rms}$ of it is defined as
\[f_{rms}=\sqrt{\lim_{n\to\infty}\frac{1}{2n}\int_{-n}^n(f(x))^2\dd{x}}.\]
